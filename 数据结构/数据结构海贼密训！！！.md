### 顺序表

顺序表每个位置可以存储**任意类型**的数据

顺序表中**重要信息**：1.size：位数 2.length：第几位 3.data_type:存储的数据类型



**增删改查**

**查找操作：可以使用并行 来较快查找速度， (每个进程负责一部分数据)**

```c
#include <stdio.h>
#include <stdlib.h>
#include <time.h>

//顺序表结构定义
typedef struct Vector {
    int *data;
    int size, length;
} Vector;
//初始化
Vector *init(int n) {
    Vector *p = (Vector *)malloc(sizeof(Vector));
    p->data = (int *)malloc(sizeof(int) * n);
    p->size = n;
    p->length = 0;
    return p;
}
//malloc 分配存储空间
//calloc 分配存储空间并初始化为0
//realloc 重新分配存储空间(先看是否可以在后面开空间) 将原先内容复制，销毁原先的空间free
//new --- delete C++不可以和malloc同用
//malloc -- free C不可以和new同用

int expand(Vector *v) {
    int extra_size = v->size;
    int *p;
    while (extra_size) {
        p = (int *)realloc(v->data, sizeof(int) * (v->size + extra_size));
        if (p) break;
        extra_size /= 2;
    }
    if (extra_size == 0) return 0;
    v->data = p;
    v->size += extra_size;
    return 1;
}
//插入
int insert(Vector *v, int ind, int val) {
    if (ind < 0 || ind > v->length) return 0;
    if (v->length == v->size) {
        if (!expand(v)) {
            return 0;
        }
    }
    for (int i = v->length - 1; i >= ind; i--) {
        v->data[i + 1] = v->data[i];
    }
    v->data[ind] = val;
    v->length += 1;
    return 1;
}
//删除
int erase(Vector *v, int ind) {
    if (ind < 0 || ind >= v->length) return 0;
    for (int i = ind + 1; i < v->length; i++) {
        v->data[i - 1] = v->data[i];
    }
    v->length -= 1;
    return 1;
}

//清空
void clear(Vector *v) {
    if (v == NULL) return ; 
    free(v->data);
    return ;
}
//输出
void output(Vector *v) {
    printf("arr = [");
    for (int i = 0; i < v->length; i++) {
        printf(" %d", v->data[i]);
    }
    printf("]\n");
    return ;
}

int main() {
    #define MAX_OP 20
    int op, ind, val;
    Vector *v = init(MAX_OP);
    for (int i = 0; i < MAX_OP; i++) {
        op = rand() % 4;
        ind = rand() % (v->length + 3) - 1;
        val = rand() % 100;
        switch (op) {
            case 0:
            case 1:
            case 2: {
                printf("insert %d to vector at %d = %d\n", val, ind, insert(v, ind, val));
                output(v);
            } break;
            case 3: {
                printf("erase element at %d from vector = %d\n", ind, erase(v, ind));
                output(v);
            } break;
            default: fprintf(stderr, "wrong op %d\n", op);
            break;
        }

    }
    clear(v);
    return 0;
}
```
